#include <stdio.h>
#include <string.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "driver/adc.h"
#include "driver/ledc.h"
#include "esp_wifi.h"
#include "esp_event.h"
#include "nvs_flash.h"
#include "esp_http_client.h"

// ====== WIFI CONFIG ======
#define WIFI_SSID "D_Link"
#define WIFI_PASS "19216812"

// ====== SERVER ======
#define SERVER_URL "http://127.0.0.1:5500/index.html"  // ⚠️ Remplace par ton IP

// ====== Pins ======
#define PIR_PIN    GPIO_NUM_13
#define LED_PIN    GPIO_NUM_12
#define LDR_CHANNEL ADC1_CHANNEL_6
#define TEMP_CHANNEL ADC1_CHANNEL_7

// ====== PWM ======
#define LEDC_TIMER LEDC_TIMER_0
#define LEDC_MODE LEDC_HIGH_SPEED_MODE
#define LEDC_CHANNEL LEDC_CHANNEL_0
#define LEDC_DUTY_RES LEDC_TIMER_12_BIT
#define LEDC_FREQUENCY 5000

// ====== Map function ======
int map_value(int x, int in_min, int in_max, int out_min, int out_max) {
    if(x < in_min) x = in_min;
    if(x > in_max) x = in_max;
    return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
}

// ====== WIFI INIT ======
void wifi_init()
{
    nvs_flash_init();
    esp_netif_init();
    esp_event_loop_create_default();

    esp_netif_create_default_wifi_sta();

    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    esp_wifi_init(&cfg);

    wifi_config_t wifi_config = {
        .sta = {
            .ssid = WIFI_SSID,
            .password = WIFI_PASS
        },
    };

    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_set_config(WIFI_IF_STA, &wifi_config);
    esp_wifi_start();
    esp_wifi_connect();

    printf("Connexion au WiFi...\n");
}
// ====== SEND DATA ======
void send_data(int ldr, int temp, int motion, int brightness)
{
    char post_data[200];

    sprintf(post_data,
        "{\"ldr\":%d,\"temp\":%d,\"motion\":%d,\"brightness\":%d}",
        ldr, temp, motion, brightness);

    esp_http_client_config_t config = {
        .url = SERVER_URL,
        .method = HTTP_METHOD_POST,
    };

    esp_http_client_handle_t client = esp_http_client_init(&config);

    esp_http_client_set_header(client, "Content-Type", "application/json");
    esp_http_client_set_post_field(client, post_data, strlen(post_data));

    esp_http_client_perform(client);
    esp_http_client_cleanup(client);
}

// ====== MAIN ======
void app_main(void)
{
    wifi_init();

    // PWM LED
    ledc_timer_config_t ledc_timer = {
        .speed_mode = LEDC_MODE,
        .timer_num = LEDC_TIMER,
        .duty_resolution = LEDC_DUTY_RES,
        .freq_hz = LEDC_FREQUENCY,
        .clk_cfg = LEDC_AUTO_CLK
    };
    ledc_timer_config(&ledc_timer);

    ledc_channel_config_t ledc_channel = {
        .speed_mode = LEDC_MODE,
        .channel = LEDC_CHANNEL,
        .timer_sel = LEDC_TIMER,
        .gpio_num = LED_PIN,
        .duty = 0
    };
    ledc_channel_config(&ledc_channel);

    // PIR
    gpio_set_direction(PIR_PIN, GPIO_MODE_INPUT);

    // ADC
    adc1_config_width(ADC_WIDTH_BIT_12);
    adc1_config_channel_atten(LDR_CHANNEL, ADC_ATTEN_DB_11);
    adc1_config_channel_atten(TEMP_CHANNEL, ADC_ATTEN_DB_11);

    int target_brightness = 0;
    int current_brightness = 0;

    while (1)
    {
        int ldr_value = adc1_get_raw(LDR_CHANNEL);
        int temp_value = adc1_get_raw(TEMP_CHANNEL);
        int motion = gpio_get_level(PIR_PIN);

        if (motion == 1)
        {
            target_brightness = map_value(ldr_value, 0, 4095, 2000, 4095);
        }
        else
        {
            target_brightness = map_value(ldr_value, 0, 4095, 0, 1500);
        }

        // Lissage
        if(current_brightness < target_brightness) current_brightness += 50;
        if(current_brightness > target_brightness) current_brightness -= 50;

        if(current_brightness < 0) current_brightness = 0;
        if(current_brightness > 4095) current_brightness = 4095;

        ledc_set_duty(LEDC_MODE, LEDC_CHANNEL, current_brightness);
        ledc_update_duty(LEDC_MODE, LEDC_CHANNEL);

        // 🔥 ENVOI AU SERVEUR
        send_data(ldr_value, temp_value, motion, current_brightness);

        printf("Data envoyée\n");

        vTaskDelay(pdMS_TO_TICKS(2000));
    }
}
